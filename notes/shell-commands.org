#+begin_src sh

  ls -l -h
  # List directories by modification time, newest first.
  ls -t

  stat foo.txt
  # List size, created, and modified timestamps for a directory.
  stat foo
  # List directory and file tree.
  tree

  tree -a
  # List directory tree.
  tree -d

  cd foo

  cd
  # Go to home directory.
  cd ~
  # Go to last directory.
  cd -

  pushd foo

  popd

** File Permissions

|   | Permission               | rwx | Binary |
|---+--------------------------+-----+--------|
| 7 | read, write, and execute | rwx |    111 |
| 6 | read and write           | rw- |    110 |
| 5 | read and execute         | r-x |    101 |
| 4 | read only                | r-- |    100 |
| 3 | write and execute        | -wx |    011 |
| 2 | write only               | -w- |    010 |
| 1 | execute only             | --x |    001 |
| 0 | none                     | --- |    000 |

Side Note: *execute* means "enter a directory".

- ~u~: User
- ~g~: Group
- ~o~: Others
- ~a~: All of the above

#+begin_src sh
  # List file permissions.
  ls -l /foo.sh
  # Add 1 to the user permission.
  chmod +100 foo.sh
  # Subtract 1 from the user permission.
  chmod -100 foo.sh
  # Give the user execute permission.
  chmod u+x foo.sh
  # Give the group execute permission.
  chmod g+x foo.sh
  # Take away the user and group execute permission.
  chmod u-x,g-x foo.sh
  # Give everybody execute permission.
  chmod u+x,g+x,o+x foo.sh
  # Give everybody execute permission.
  chmod a+x foo.sh
  # Give everybody execute permission.
  chmod +x foo.sh
#+end_src

** Find Files

Find the binary files for a command.

#+begin_src sh
  # Find the binary.
  type wget
  # Find the binary.
  which wget
  # Find the binary, source, and manual page files.
  whereis wget
#+end_src

~locate~ uses an index and is fast.

#+begin_src sh
  # Update the index.
  updatedb
  # Find "foo.txt".
  locate foo.txt
  # Find a file while ignoring case.
  locate --ignore-case "foo.txt"
  # Find a text file starting with "f".
  locate f*.txt
#+end_src

~find~ doesn't use an index and is slow.

#+begin_src sh
  # Find a file.
  find /path -name foo.txt
  # Find a file with case insensitive search.
  find /path -iname foo.txt
  # Find all text files.
  find /path -name "*.txt"
  # Find a file and delete it.
  find /path -name foo.txt -delete
  # Find all .png files and execute pngquant on it
  find /path -name "*.png" -exec pngquant {}
  # Find a file.
  find /path -type f -name foo.txt
  # Find a directory.
  find /path -type d -name foo
  # Find a symbolic link.
  find /path -type l -name foo.txt
  # Find files that haven't been modified in 30 days.
  find /path -type f -mtime +30
  # Delete files that haven't been modified in 30 days.
  find /path -type f -mtime +30 -delete
#+end_src

** Find in Files

#+begin_src sh
  # Search for "foo" in file "bar.txt".
  grep 'foo' /bar.txt
  # Search for "foo" in directory "bar".
  # grep "foo" /bar --recursive
  grep "foo" /bar -r
  # Search for "foo" in directory "bar" and follow symbolic links.
  # grep "foo" /bar --dereference-recursive
  grep "foo" /bar -R
  # Show only files that match.
  # grep "foo" /bar --files-with-matches
  grep "foo" /bar -l
  # Show only files that don't match.
  # grep "foo" /bar --files-without-match
  grep "foo" /bar -L
  # Case insensitive search.
  # grep "Foo" /bar --ignore-case
  grep "Foo" /bar -i
  # Match the entire line.
  # grep "foo" /bar --line-regexp
  grep "foo" /bar -x
  # Add N lines of context above and below each search result.
  # grep "foo" /bar --context 1
  grep "foo" /bar -C 1
  # Show only lines that don't match.
  # grep "foo" /bar --invert-match
  grep "foo" /bar -v
  # Count the number lines that match.
  # grep "foo" /bar --count
  grep "foo" /bar -c
  # Add line numbers.
  # grep "foo" /bar --line-number
  grep "foo" /bar -n
  # Add color to output.
  grep "foo" /bar --colour
  # Search for "foo" or "bar" in directory "baz".
  grep "foo|bar" /baz -R
  # Use extended regular expressions.
  # grep --extended-regexp "foo|bar" /baz -R
  grep -E "foo|bar" /baz -R
  # Use regular expressions.
  egrep "foo|bar" /baz -R
#+end_src

** Replace in Files

#+begin_src sh
  # Replace "fox" with "bear" in "foo.txt". Output to console.
  sed "s/fox/bear/g" foo.txt
  # Replace "fox" (case insensitive) with "bear" in "foo.txt". Output to console.
  sed "s/fox/bear/gi" foo.txt
  # Replace "red" with "blue" and "fox" with "bear" in "foo.txt". Output to console.
  sed "s/red fox/blue bear/g" foo.txt
  # Replace "fox" with "bear" in "foo.txt". Save in "bar.txt".
  sed "s/fox/bear/g" foo.txt > bar.txt
  # Replace "fox" with "bear". Overwrite "foo.txt".
  # sed "s/fox/bear/g" foo.txt --in-place
  sed "s/fox/bear/g" foo.txt -i
#+end_src

** Compress Files

*** ~zip~

Compresses one or more files into *.zip files.

#+begin_src sh
  # Compress "bar.txt" into "foo.zip".
  zip foo.zip /bar.txt
  # Compress "bar.txt" and "baz.txt" into "foo.zip".
  zip foo.zip /bar.txt /baz.txt
  # Compress "bar.txt" and "baz.txt" into "foo.zip".
  zip foo.zip /{bar,baz}.txt
  # Compress directory "bar" into "foo.zip".
  # zip --recurse-paths foo.zip /bar
  zip -r foo.zip /bar
#+end_src

*** ~gzip~

Compresses a single file into a *.gz file.

#+begin_src sh
  # Compress "bar.txt" into "foo.gz" and then delete "bar.txt".
  gzip /bar.txt foo.gz
  # Compress "bar.txt" into "foo.gz".
  # gzip --keep /bar.text foo.gz
  gzip -k /bar.txt foo.gz
#+end_src

*** ~tar -c~

Combines and optionally compresses one or more files into a single
*.tar, *.tar.gz, *.tpz, or *.tgz file.

#+begin_src sh
  # Compress "bar.txt" and "baz.txt" into "foo.tgz".
  # tar --create --gzip --file=foo.tgz /bar.txt /baz.txt
  tar -c -z -f=foo.tgz /bar.txt /baz.txt
  # Compress "bar.txt" and "baz.txt" into "foo.tgz".
  # tar --create --gzip --file=foo.tgz /{bar,baz}.txt
  tar -c -z -f=foo.tgz /{bar,baz}.txt
  # Compress directory "bar" into "foo.tgz".
  # tar --create --gzip --file=foo.tgz /bar
  tar -c -z -f=foo.tgz /bar
#+end_src

** Multiplex Terminals

Start multiple terminal sessions. Active sessions persist reboots.
~tmux~ is the more modern equivalent of ~screen~.

#+begin_src sh
  # Start a new session (CTRL-b + d to detach).
  tmux
  # List all sessions.
  tmux ls
  # Reattach to a session
  tmux attach -t 0
   # Start a new session (CTRL-a + d to detach).
  screen
  # List all sessions.
  screen -ls
  # Reattach to a session.
  screen -R 31166
  # Exit a session.
  exit
#+end_src

#+begin_src sh

*** Conditional Statements

**** Boolean Operators

- ~$foo~: truthy
- ~!$foo~: falsy

**** Numeric Operators

- ~-eq~: Equals
- ~-ne~: Not equals
- ~-gt~: Greater than
- ~-ge~: Greater than or equal to
- ~-lt~: Less than
- ~-le~: Less than or equal to
- ~-e foo.txt~: Check file exists
- ~-z foo~: Check if variable exists

**** String Operators

- ~=~: Equals
- ~==~: Equals
- ~-z~: Is null
- ~-n~: Is not null
- ~<~: Is less than in ASCII alphabetical order
- ~>~: Is greater than in ASCII alphabetical order
